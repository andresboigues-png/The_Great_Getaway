# Trip-PDF export (2026-05-18). reportlab is pure-Python — no native
# dep, resolves on PA's free-tier `pip install` without surprises.
# xhtml2pdf would have given us HTML-template rendering but it pulls
# `pycairo` as a transitive build dep, which needs the cairo system
# library at build time (PA can't install that). reportlab is more
# code per page but produces higher-fidelity PDFs and ships fonts
# cleanly. The trip-PDF builder lives at routes/pdf.py.
reportlab==5.0.0

# R3-Fix #17: Pillow is imported directly by both routes/media.py
# (EXIF strip on photo uploads) and routes/pdf.py (Static Maps
# image sizing). Reportlab pulls Pillow as a transitive dep today
# but a future resolver pass could drop it on a wheel build that
# doesn't need PIL — which would silently break:
#   - photo uploads (EXIF strip falls back to bytes-verbatim save
#     → GPS lat/lng + camera serial preserved in every shared photo);
#   - PDF map sizing (would crash on `from PIL import Image`).
# Pin explicitly so the install is deterministic regardless of
# reportlab's resolution.
Pillow==12.3.0

# R3-Round 4 fix: HEIC/HEIF decoder so iPhone "Most Compatible"
# off uploads (the default for newer iPhones) flow through PIL's
# EXIF strip pipeline. Without this, HEIC uploads were rejected at
# /api/upload with a "switch to JPEG" message (R3-Round 3 B1). With
# pillow-heif registered, iOS users no longer have to change their
# camera setting — HEIC uploads work end-to-end and get EXIF
# stripped just like JPEG.
pillow-heif==1.4.0
